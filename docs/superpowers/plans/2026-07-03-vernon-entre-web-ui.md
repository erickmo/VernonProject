# vernon_entre Student Web UI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A responsive, installable React SPA served at `/entre` where a student logs in, manages Ventures, and fills each venture's four canvases (SWOT, BMC, VPC, Empathy Map) as sticky-note boards, reading/writing the `vernon_entre` doctypes via Frappe's generic client API.

**Architecture:** New Vite+React+TS app `frontend-entre/` mirroring vernon_project's serving convention (build → `public/frontend_entre/` → `www/entre.html`, `BrowserRouter basename="/entre"`, `website_route_rules`). One generic config-driven `CanvasEditor` renders all four canvases from `configs.ts`. Data via `frappe.client.*` (no custom Python); `if_owner` doctype perms enforce per-student isolation. Installable PWA (hand-written service worker + manifest scoped `/entre`). Playful own visual identity via the frontend-design skill.

**Tech Stack:** Vite, React 18, TypeScript, React Router v6, TailwindCSS v3, Vitest + @testing-library/react. Frappe session-cookie auth.

## Global Constraints

- App path: SPA mounts at `/entre`; router `basename="/entre"`.
- Build base: `--base=/assets/vernon_entre/frontend_entre/`; Vite `build.outDir: '../vernon_entre/public/frontend_entre'`.
- Served shell: `vernon_entre/www/entre.html` (generated by copy-html) + `vernon_entre/www/entre.py` with `no_cache = 1`.
- Route rule (hooks.py): `{"from_route": "/entre/<path:app_path>", "to_route": "entre"}`.
- API: use ONLY Frappe built-ins (`frappe.client.get_list/get/insert/save/set_value/delete`, `frappe.auth.get_logged_user`, `login`). NO custom `api/*.py`. Base `/api/method/`; GET no CSRF; POST header `X-Frappe-CSRF-Token: window.csrf_token`; every request `credentials: 'same-origin'`; unwrap response `message`.
- Doctypes (exist, live on dev.vernon.id): `Venture` (venture_name, pitch, status Draft/Active/Archived), `SWOT`, `Business Model Canvas`, `Value Proposition Canvas`, `Empathy Map`; child `Entre Canvas Item` (item, note, priority High/Medium/Low default Medium). Student = owner; role `Entre Student`.
- No native `alert()/confirm()/prompt()` — use in-app dialog/toast (project rule).
- Item editing: sticky cards, add/edit/delete, priority badge, NO drag. Explicit Save button (no autosave in v1).
- Build/verify site: `dev.vernon.id`. Do NOT touch production.
- Work happens in the `vernon_entre` app repo (`/home/frappe/frappe-bench/apps/vernon_entre`), branch `develop`; commit there.

```ts
// api.ts
const METHOD = '/api/method/'
const csrf = () => (window as any).csrf_token || ''

async function get(method: string, params: Record<string, any> = {}) {
  const qs = new URLSearchParams()
  for (const [k, v] of Object.entries(params)) qs.set(k, typeof v === 'string' ? v : JSON.stringify(v))
  const res = await fetch(`${METHOD}${method}?${qs.toString()}`, { credentials: 'same-origin' })
  if (!res.ok) throw new Error(`${res.status}`)
  return (await res.json()).message
}
async function post(method: string, body: Record<string, any> = {}) {
  const res = await fetch(`${METHOD}${method}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'X-Frappe-CSRF-Token': csrf() },
    body: JSON.stringify(body),
    credentials: 'same-origin',
  })
  if (!res.ok) throw new Error(`${res.status}`)
  return (await res.json()).message
}

export const getList = <T = any>(doctype: string, opts: { filters?: Record<string, any>; fields?: string[]; limit?: number } = {}): Promise<T[]> =>
  get('frappe.client.get_list', { doctype, filters: opts.filters ?? {}, fields: opts.fields ?? ['name'], limit_page_length: opts.limit ?? 0 })
export const getDoc = <T = any>(doctype: string, name: string): Promise<T> => get('frappe.client.get', { doctype, name })
export const insertDoc = <T = any>(doc: any): Promise<T> => post('frappe.client.insert', { doc })
export const saveDoc = <T = any>(doc: any): Promise<T> => post('frappe.client.save', { doc })
export const setValue = (doctype: string, name: string, fieldname: string, value: any): Promise<void> => post('frappe.client.set_value', { doctype, name, fieldname, value })
export const deleteDoc = (doctype: string, name: string): Promise<void> => post('frappe.client.delete', { doctype, name })
export const whoami = (): Promise<string> => get('frappe.auth.get_logged_user')
export const login = async (usr: string, pwd: string): Promise<void> => {
  const res = await fetch(`${METHOD}login`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ usr, pwd }), credentials: 'same-origin' })
  if (!res.ok) throw new Error('Invalid login')
}
```

```tsx
import { createContext, useContext, useEffect, useState, ReactNode } from 'react'
import { whoami } from './lib/api'
import Login from './pages/Login'

const UserCtx = createContext<string | null>(null)
export const useUser = () => useContext(UserCtx)

export function RequireAuth({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<string | null | undefined>(undefined)
  useEffect(() => { whoami().then(u => setUser(u && u !== 'Guest' ? u : null)).catch(() => setUser(null)) }, [])
  if (user === undefined) return <div className="p-8">Loading…</div>
  if (user === null) return <Login onDone={() => { window.location.href = '/entre' }} />
  return <UserCtx.Provider value={user}>{children}</UserCtx.Provider>
}
```

```tsx
import { useState } from 'react'
import { login } from '../lib/api'
import { Button } from '../ui/Button'

export default function Login({ onDone }: { onDone: () => void }) {
  const [usr, setUsr] = useState(''); const [pwd, setPwd] = useState(''); const [err, setErr] = useState('')

    e.preventDefault(); setErr('')

  return (
    <form onSubmit={submit} className="max-w-sm mx-auto p-8 space-y-3">
      <label className="block">Email<input aria-label="Email" value={usr} onChange={e => setUsr(e.target.value)} className="border w-full" /></label>
      <label className="block">Password<input aria-label="Password" type="password" value={pwd} onChange={e => setPwd(e.target.value)} className="border w-full" /></label>
      {err && <p role="alert" className="text-red-600">{err}</p>}
```

```tsx
import { useState } from 'react'
import { CanvasItem, Priority } from '../lib/types'
import { SectionDef } from './configs'
import { ItemCard } from './ItemCard'
import { Dialog } from '../ui/Dialog'
import { Button } from '../ui/Button'

export function CanvasSection({ def, items, onChange }: { def: SectionDef; items: CanvasItem[]; onChange: (next: CanvasItem[]) => void }) {
  const [editing, setEditing] = useState<number | null>(null) // index, or -1 for new
  const [text, setText] = useState(''); const [note, setNote] = useState(''); const [priority, setPriority] = useState<Priority>('Medium')
  const open = (i: number | null) => {
    if (i !== null && i >= 0) { const it = items[i]; setText(it.item); setNote(it.note ?? ''); setPriority(it.priority) }
    else { setText(''); setNote(''); setPriority('Medium') }
    setEditing(i === null ? -1 : i)
  }
  const save = () => {
    if (!text.trim()) return
    const next = [...items]; const row: CanvasItem = { item: text.trim(), note: note.trim() || undefined, priority }
    if (editing === -1) next.push(row); else if (editing !== null) next[editing] = { ...next[editing], ...row }
    onChange(next); setEditing(null)
  }
  const del = (i: number) => onChange(items.filter((_, j) => j !== i))
  return (
    <section style={{ gridArea: def.fieldname }} className="flex flex-col gap-2 p-2 min-h-[8rem]">
      <h3 className="font-semibold text-sm">{def.label}</h3>
      {items.map((it, i) => <ItemCard key={i} item={it} onEdit={() => open(i)} onDelete={() => del(i)} />)}
      <button aria-label={`add to ${def.label}`} onClick={() => open(null)} className="text-sm opacity-70 self-start">+ add</button>
      <Dialog open={editing !== null} onClose={() => setEditing(null)} title={def.label}>
        <input aria-label="new item text" value={text} onChange={e => setText(e.target.value)} className="border w-full" />
        <input aria-label="note" value={note} onChange={e => setNote(e.target.value)} placeholder="note (optional)" className="border w-full mt-2" />
        <select aria-label="priority" value={priority} onChange={e => setPriority(e.target.value as Priority)} className="border mt-2">
          <option>High</option><option>Medium</option><option>Low</option>
        </select>
        <Button onClick={save}>Save item</Button>
      </Dialog>
    </section>
  )
}
```

- [ ] **Step 5: Implement `CanvasEditor.tsx`** (container: load, hold doc state, save)
```tsx
import { useEffect, useState } from 'react'
import { CanvasConfig } from './configs'
import { CanvasItem } from '../lib/types'
import { getList, getDoc, saveDoc } from '../lib/api'
import { CanvasSection } from './CanvasSection'
import { Button } from '../ui/Button'
import { useToast } from '../ui/Toast'

type DocState = { name?: string; venture: string; doctype: string; [field: string]: any }

export default function CanvasEditor({ config, venture }: { config: CanvasConfig; venture: string }) {
  const [doc, setDoc] = useState<DocState | null>(null)
  const [dirty, setDirty] = useState(false)
  const toast = useToast()
  useEffect(() => {
    let live = true
    getList(config.doctype, { filters: { venture }, fields: ['name'], limit: 1 }).then(async rows => {
      const base: DocState = { doctype: config.doctype, venture }
      config.sections.forEach(s => { base[s.fieldname] = [] as CanvasItem[] })
      if (rows[0]?.name) { const full = await getDoc(config.doctype, rows[0].name); if (live) setDoc({ ...base, ...full }) }
      else if (live) setDoc(base)
    })
    return () => { live = false }
  }, [config.doctype, venture])
  if (!doc) return <div className="p-6">Loading…</div>
  const setSection = (field: string, items: CanvasItem[]) => { setDoc({ ...doc, [field]: items }); setDirty(true) }
  const save = async () => {
    try { const saved = await saveDoc<DocState>(doc); setDoc({ ...doc, name: saved.name }); setDirty(false); toast.show('Saved', 'ok') }
    catch { toast.show('Save failed', 'error') }
  }
  return (
    <div className="p-4">
      <div className="flex items-center justify-between mb-3">
        <h1 className="text-xl">{config.title}</h1>
        <Button onClick={save} disabled={!dirty}>Save</Button>
      </div>
      <div className="grid gap-3 entre-canvas" style={{ gridTemplateColumns: config.gridCols, gridTemplateAreas: config.gridAreas }}>
        {config.sections.map(s => <CanvasSection key={s.fieldname} def={s} items={doc[s.fieldname] ?? []} onChange={items => setSection(s.fieldname, items)} />)}
      </div>
      {toast.Host}
    </div>
  )
}
```

```js
const ASSET_PREFIX = '/assets/vernon_entre/frontend_entre/'
const SHELL = '/entre'
const CACHE = 'entre-v1'
self.addEventListener('activate', e => e.waitUntil(caches.keys().then(ks => Promise.all(ks.filter(k => k !== CACHE).map(k => caches.delete(k))))))
self.addEventListener('fetch', e => {
  const url = new URL(e.request.url)
  if (url.pathname.startsWith(ASSET_PREFIX)) {
    e.respondWith(caches.open(CACHE).then(c => c.match(e.request).then(r => r || fetch(e.request).then(res => { c.put(e.request, res.clone()); return res }))))
  } else if (e.request.mode === 'navigate' && url.pathname.startsWith(SHELL)) {
    e.respondWith(fetch(e.request).catch(() => caches.match(SHELL)))
  }
})
```
